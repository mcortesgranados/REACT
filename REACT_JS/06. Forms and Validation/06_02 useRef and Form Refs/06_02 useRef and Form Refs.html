06_02 useRef and Form Refs - Part 1

Q1: What is useRef and how does it work in React?

useRef is a React Hook that returns a mutable ref object whose .current property is initialized to the passed argument. This ref object persists for the full lifetime of the component. 

import React, { useRef } from 'react';

function RefExample() {
  const inputRef = useRef(null); // Create ref

  const focusInput = () => {
    // Access DOM element directly
    inputRef.current.focus();
  };

  return (
    <div>
      <input ref={inputRef} type="text" placeholder="Click button to focus" />
      <button onClick={focusInput}>Focus Input</button>
    </div>
  );
}

export default RefExample;

Q3: Can useRef hold any value, not just DOM nodes?

Yes. useRef can hold any mutable value, not just DOM references.

import React, { useEffect, useRef } from 'react';

function TimerCounter() {
  const count = useRef(0);

  useEffect(() => {
    const interval = setInterval(() => {
      count.current += 1; // Mutate without causing re-render
      console.log("Count:", count.current);
    }, 1000);

    return () => clearInterval(interval);
  }, []);

  return <div>Open console to see counter</div>;
}

export default TimerCounter;

function FocusSwitcher() {

  const lastNameRef = useRef(null);

  const switchFocus = () => {

      lastNameRef.current.focus();

      <input ref={firstNameRef} placeholder="First Name" /><br />

Q50: How to implement a focus trap using useRef in React?

useRef can be used to implement a focus trap that ensures focus stays within a particular area of the page (e.g., modal).

import React, { useRef, useEffect } from 'react';

function FocusTrap() {
  const firstInputRef = useRef(null);
  const lastInputRef = useRef(null);

  useEffect(() => {
    const handleTab = (e) => {
      if (e.key === 'Tab') {
        if (document.activeElement === lastInputRef.current) {
          firstInputRef.current.focus();
          e.preventDefault();
        }
      }
    };

    document.addEventListener('keydown', handleTab);
    return () => {
      document.removeEventListener('keydown', handleTab);
    };
  }, []);

  return (
    <div>
      <input ref={firstInputRef} type="text" placeholder="First Input" /><br />
      <input type="text" placeholder="Second Input" /><br />
      <input ref={lastInputRef} type="text" placeholder="Last Input" />
    </div>
  );
}
